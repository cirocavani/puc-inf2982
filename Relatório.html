function optret(xopt, name)
    r = nothing
    for i=1:n
        xopt[i] == 0.0 && continue
        q = assets[i]
        s = df[df[:CODNEG] .== q, :][:RET] * xopt[i]
        if r === nothing
            r = s
        else
            r += s
        end
    end
    DataFrame(DATE = unique(df[:DATE]), CODNEG = fill(name, length(r)), RET = r)
end
